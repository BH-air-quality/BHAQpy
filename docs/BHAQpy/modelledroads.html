470def _calc_gradient_percentage(distances, heightsAOD):
471
472    # calculate a gradient as percentage using dtm height
473    distances = [i.value() if type(i) == QVariant else float(i) for i in distances]
474    heightsAOD = [i.value() if type(i) == QVariant else float(i) for i in heightsAOD]
475    
476    heightsAOD_2 = []
477    for i in heightsAOD:
478        if type(i) == QVariant:
479            #print(type(i.value()))
480            heightsAOD_2.append(NULL)
481        else:
482            heightsAOD_2.append(float(i))
483    
484    distances_2 = []
485    for i in distances:
486        if type(i) == QVariant:
487            #print(type(i.value()))
488            distances_2.append(NULL)
489        else:
490            distances_2.append(float(i))
491    
492    rises = abs(np.diff(heightsAOD_2))
493    runs = abs(np.diff(distances_2))
494    percentages = pd.DataFrame({'rise' : rises, 'run' : runs})
495    percentages['percentage'] = (percentages['rise']/percentages['run'])*100
496
497    avg_percentage = np.mean(percentages['percentage']) 
498    
499    return avg_percentage

501def _init_modelled_roads_layer(input_modelled_road_layer, gpkg_write_path, gpkg_layer_name,

548def _copy_input_layer_geometry(modelled_road_layer, input_modelled_road_layer, 
549                               traffic_count_point_id_col_name, width_col_name, 
550                               speed_col_name, junction_col_name, road_height_col_name,
551                               canyon_height_col_name):
552    
553    original_layer_fields = [field.name() for field in input_modelled_road_layer.fields()]
554    
555    if traffic_count_point_id_col_name not in original_layer_fields:
556        raise Exception((f"traffic_count_point_id_col_name: {traffic_count_point_id_col_name}"
557                        "not found in specified layer"))
558    
559    default_values = {width_col_name : 0, speed_col_name : 0, junction_col_name : False, 
560                      road_height_col_name : 0, canyon_height_col_name : 0}
561    
562    with edit(modelled_road_layer):
563        
564        dp = modelled_road_layer.dataProvider()
565        
566        tcp_register = []
567        #look through each feature - copy the geometry but change the attributes
568        for original_feature in input_modelled_road_layer.getFeatures():
569            
570            original_tcp_id = original_feature[traffic_count_point_id_col_name]
571            
572            original_values = []
573            for col_name in list(default_values.keys()):
574                if col_name in original_layer_fields and original_feature[col_name] != NULL:
575                    original_value = original_feature[col_name]
576                else:
577                    original_value = default_values[col_name]
578                
579                original_values.append(original_value)
580            
581            original_width, original_speed, original_junction, original_height, original_canyon_height = original_values
582            
583            #skip if no id
584            if  original_tcp_id == NULL or original_tcp_id == '':
585                continue
586            
587            # initiate feture
588            new_feature = QgsFeature()
589            # copy geom
590            if original_feature.geometry().isNull() or original_feature.geometry().isEmpty():
591                continue
592            
593            new_feature.setGeometry(original_feature.geometry())
594            
595            # get source id
596            if original_junction == True:
597                junction_str = '.J'
598            else:
599                junction_str = ''
600            
601            tcp_id = str(original_tcp_id)+junction_str
602            number = len([i for i in tcp_register if i == tcp_id]) + 1
603            source_id = str(tcp_id) + '.' + str(number)  
604            # tracker
605            tcp_register.append(tcp_id) 
606            
607            #create feature
608            new_feature.setAttributes([None, source_id, str(original_tcp_id), 
609                                       original_junction,
610                                       original_width, original_speed,
611                                       0, original_height, original_canyon_height])
612            #add feature   
613            dp.addFeatures([new_feature])
614
